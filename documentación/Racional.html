def initialize(num, denom)

  raise TypeError, "El denominador debe ser mayor que cero" unless (denom != 0)

  div = gcd(num, denom)
  @num = num/div
  @denom = denom/div

end

def <=>(o)
  return nil unless o.instance_of? Racional
  (num.to_f / denom) <=> (o.num.to_f / o.denom)
end
